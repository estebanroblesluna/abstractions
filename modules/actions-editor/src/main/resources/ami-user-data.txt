#!/bin/bash
if [ -f "/usr/local/jetty-9.1.3/webapps/server.war" ]
then
	echo "Server found, no need to download"
else
	echo "Downloading server..."
	cd /usr/local/jetty-9.1.3/webapps/
	wget http://www.abztractionz.com/wars/latest/server.war
	unzip server.war -d server/
	rm server.war
	echo "Server downloaded"
fi

if [-f "/usr/local/abstractions-apps"]
then
	echo "Abstractions app directory already there"
else
	echo "Creating abstractions app directory..."
	mkdir /usr/local/abstractions-apps
	chmod 777 /usr/local/abstractions-apps
	echo "Abstractions app directory created"
fi

sed -i "s|server.apps.directory=[^ ]*|server.apps.directory=$$$server.apps.directory$$$|g" /usr/local/jetty-9.1.3/webapps/server/WEB-INF/classes/server.properties
sed -i "s|server.editor.url=[^ ]*|server.editor.url=$$$server.editor.url$$$|g" /usr/local/jetty-9.1.3/webapps/server/WEB-INF/classes/server.properties
sed -i "s|server.id=[^ ]*|server.id=$$$server.id$$$|g" /usr/local/jetty-9.1.3/webapps/server/WEB-INF/classes/server.properties
sed -i "s|server.key=[^ ]*|server.key=$$$server.key$$$|g" /usr/local/jetty-9.1.3/webapps/server/WEB-INF/classes/server.properties
